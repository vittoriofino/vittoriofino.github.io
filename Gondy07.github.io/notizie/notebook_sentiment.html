    """
    Analizza il sentiment di un testo usando un modello LLM locale

                1 = negativo

                3 = molto positivo
                
                Testo: {text}
                
                Numero:"""
    
    data = {
        "model": "granite-3.0-2b-instruct",
        "prompt": prompt,
        "temperature": 0.0,
        "max_tokens": 1,
        "stream": False
    }
    
    try:
        response = requests.post(api_url, headers=headers, data=json.dumps(data))
        response.raise_for_status()
        return response.json()
    except:
        return None

        if response_text in ['0', '1', '2', '3']:
            return int(response_text)

    except:
        return 1

import pandas as pd
import requests
import json
import time

def summarize_text(text, api_url="http://localhost:1234/api/v0/completions", max_attempts=5):
    """
    Summarizes the text using a local LLM model
    """
    headers = {
        "Content-Type": "application/json"
    }
    
    prompt = f"""Summarize the following text in exactly ONE word (not 'Crime'), without articles:
                Text: {text}
                
                One-word summary:"""
    
    data = {
        "model": "granite-3.0-2b-instruct",
        "prompt": prompt,
        "temperature": 0.0,
        "max_tokens": 1,
        "stream": False
    }
    
    attempts = 0
    while attempts < max_attempts:
        try:
            response = requests.post(api_url, headers=headers, data=json.dumps(data))
            response.raise_for_status()
            summary = response.json()['choices'][0]['text'].strip()
            if summary.lower() not in ["the", "crime"] and len(summary.split()) == 1:
                return summary
            else:
                print(f"Attempt {attempts+1}: {summary}")
        except requests.exceptions.RequestException as e:
            print(f"Error in API call: {e}")
        
        attempts += 1
        time.sleep(1)  # Attende un secondo prima di riprovare

    return "Unknown"

import pandas as pd

def merge_sentiment_data(source_file, target_file):
    """
    Copia i valori Sentiment dal file sorgente al file target basandosi sull'ID.
    
    Parameters:
    source_file: Path del file CSV contenente i valori Sentiment da copiare
    target_file: Path del file CSV dove aggiungere i valori Sentiment
    """
    # Leggi i file CSV
    source_df = pd.read_csv(source_file)
    target_df = pd.read_csv(target_file)
    
    # Crea un dizionario di ID -> Sentiment dal file sorgente
    sentiment_dict = dict(zip(source_df['ID'], source_df['Sentiment']))
    
    # Aggiungi i valori Sentiment al target_df basandosi sull'ID
    target_df['Sentiment'] = target_df['ID'].map(sentiment_dict)
    
    # Salva il risultato sovrascrivendo il file target
    target_df.to_csv(target_file, index=False)
    
    print(f"Aggiornati {len(target_df)} record nel file target.")

import pandas as pd
from wordcloud import WordCloud
import matplotlib.pyplot as plt
import numpy as np
from PIL import Image, ImageDraw, ImageFont

def create_crime_mask():
    # Crea un'immagine vuota
    width = 1000
    height = 300
    mask = Image.new('RGB', (width, height), 'white')
    draw = ImageDraw.Draw(mask)
    
    # Usa un font grande e in grassetto
    try:
        font = ImageFont.truetype("arial.ttf", 200)
    except:
        font = ImageFont.load_default()
    
    # Disegna la parola "CRIME"
    draw.text((50, 50), "CRIME", font=font, fill='black')
    
    # Converti l'immagine in array numpy
    mask_array = np.array(mask)
    
    # Crea la maschera binaria (nero = maschera)
    final_mask = mask_array[:, :, 0].astype(np.int_) == 0
    
    return final_mask

def generate_crime_wordcloud(csv_file):
    """
    Genera una word cloud a forma della parola CRIME
    """
    # Leggi il CSV
    df = pd.read_csv(csv_file)
    
    # Unisci tutti i summary in un unico testo, escludendo "Unknown"
    text = ' '.join([str(summary) for summary in df['Summary'] if str(summary).lower() != 'unknown'])
    
    # Crea la maschera
    mask = create_crime_mask()
    
    # Configura il word cloud
    wordcloud = WordCloud(
        width=800,
        height=250,
        background_color='white',
        mask=mask,
        contour_width=1,
        contour_color='black',
        min_font_size=8,
        max_font_size=80,
        collocations=False
    ).generate(text)
    
    # Crea la figura
    plt.figure(figsize=(15, 5))
    plt.imshow(wordcloud, interpolation='bilinear')
    plt.axis('off')
    
    # Salva l'immagine
    plt.savefig('crime_wordcloud.png', bbox_inches='tight', dpi=300, facecolor='white')
    plt.close()
    
    print("Word cloud creata e salvata come 'crime_wordcloud.png'")
